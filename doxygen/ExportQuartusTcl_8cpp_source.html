60 void emitPath(TclOutputState &s, PlacementDB::PlacedInstance inst) {
61  // Traverse each part of the path.
62  auto parts = inst.path.getAsRange<hw::InnerRefAttr>();
63  auto lastPart = std::prev(parts.end());
64  for (auto part : parts) {
65  emitInnerRefPart(s, part);
66  if (part != *lastPart)
67  s.os << '|';
68  }
69 
70  // Some placements don't require subpaths.
71  if (!inst.subpath.empty()) {
72  s.os << '|';
73  s.os << inst.subpath;
74  }
75 
76  s.os << '\n';
77 }

79 /// Emit tcl in the form of:
80 /// "set_location_assignment MPDSP_X34_Y285_N0 -to $parent|fooInst|entityName"
81 static void emit(TclOutputState &s, PlacementDB::PlacedInstance inst,
82  PhysLocationAttr pla) {
83 
84  s.indent() << "set_location_assignment ";
85 
86  // Different devices have different 'number' letters (the 'N' in 'N0'). M20Ks
87  // and DSPs happen to have the same one, probably because they never co-exist
88  // at the same location.
89  char numCharacter;
90  switch (pla.getPrimitiveType().getValue()) {
91  case PrimitiveType::M20K:
92  s.os << "M20K";
93  numCharacter = 'N';
94  break;
95  case PrimitiveType::DSP:
96  s.os << "MPDSP";
97  numCharacter = 'N';
98  break;
99  case PrimitiveType::FF:
100  s.os << "FF";
101  numCharacter = 'N';
102  break;
103  }
104 
105  // Write out the rest of the location info.
106  s.os << "_X" << pla.getX() << "_Y" << pla.getY() << "_" << numCharacter
107  << pla.getNum();
108 
109  // To which entity does this apply?
110  s.os << " -to $parent|";
111  emitPath(s, inst);
112 }

114 /// Emit tcl in the form of:
115 /// set_instance_assignment -name PLACE_REGION "X1 Y1 X20 Y20" -to $parent|a|b|c
116 /// set_instance_assignment -name RESERVE_PLACE_REGION OFF -to $parent|a|b|c
117 /// set_instance_assignment -name CORE_ONLY_PLACE_REGION ON -to $parent|a|b|c
118 /// set_instance_assignment -name REGION_NAME test_region -to $parent|a|b|c
119 static void emit(TclOutputState &s, PlacementDB::PlacedInstance inst,
120  PhysicalRegionRefAttr regionRef) {
121  auto topModule = inst.op->getParentOfType<mlir::ModuleOp>();
122  auto physicalRegion =
123  topModule.lookupSymbol<PhysicalRegionOp>(regionRef.getPhysicalRegion());
124  assert(physicalRegion && "must reference an existant physical region");
125 
126  // PLACE_REGION directive.
127  s.indent() << "set_instance_assignment -name PLACE_REGION \"";
128  auto physicalBounds =
129  physicalRegion.bounds().getAsRange<PhysicalBoundsAttr>();
130  llvm::interleave(
131  physicalBounds, s.os,
132  [&s](PhysicalBoundsAttr bounds) {
133  s.os << 'X' << bounds.getXMin() << ' ';
134  s.os << 'Y' << bounds.getYMin() << ' ';
135  s.os << 'X' << bounds.getXMax() << ' ';
136  s.os << 'Y' << bounds.getYMax();
137  },
138  ";");
139  s.os << '"';
140  s.os << " -to $parent|";
141  emitPath(s, inst);
142 
143  // RESERVE_PLACE_REGION directive.
144  s.indent() << "set_instance_assignment -name RESERVE_PLACE_REGION OFF";
145  s.os << " -to $parent|";
146  emitPath(s, inst);
147 
148  // CORE_ONLY_PLACE_REGION directive.
149  s.indent() << "set_instance_assignment -name CORE_ONLY_PLACE_REGION ON";
150  s.os << " -to $parent|";
151  emitPath(s, inst);
152 
153  // REGION_NAME directive.
154  s.indent() << "set_instance_assignment -name REGION_NAME ";
155  s.os << physicalRegion.getName();
156  s.os << " -to $parent|";
157  emitPath(s, inst);
158 }

160 /// Write out all the relevant tcl commands. Create one 'proc' per module which
161 /// takes the parent entity name since we don't assume that the created module
162 /// is the top level for the entire design.
163 LogicalResult circt::msft::exportQuartusTcl(MSFTModuleOp hwMod,
164  StringRef outputFile) {
165  // Build up the output Tcl, tracking symbol references in state.
166  std::string s;
167  llvm::raw_string_ostream os(s);
168  TclOutputState state(os);
169  PlacementDB db(hwMod);
170  size_t failures = db.addDesignPlacements();
171  if (failures != 0)
172  return hwMod->emitError("Could not place ") << failures << " instances";
173 
174  os << "proc " << hwMod.getName() << "_config { parent } {\n";
175 
176  db.walkPlacements(
177  [&state](PhysLocationAttr loc, PlacementDB::PlacedInstance inst) {
178  emit(state, inst, loc);
179  });
180 
181  db.walkRegionPlacements([&state](PhysicalRegionRefAttr regionRef,
182  PlacementDB::PlacedInstance inst) {
183  emit(state, inst, regionRef);
184  });
185 
186  os << "}\n\n";
187 
188  // Create a verbatim op containing the Tcl and symbol references.
189  OpBuilder builder = OpBuilder::atBlockEnd(hwMod->getBlock());
190  auto verbatim = builder.create<sv::VerbatimOp>(
191  builder.getUnknownLoc(), os.str(), ValueRange{},
192  builder.getArrayAttr(state.symbolRefs));
193 
194  // When requested, give the verbatim op an output file.
195  if (!outputFile.empty()) {
196  auto outputFileAttr =
197  OutputFileAttr::getFromFilename(builder.getContext(), outputFile);
198  verbatim->setAttr("output_file", outputFileAttr);
199  }
200 
201  return success();
202 }
