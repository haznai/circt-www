56 LogicalResult
57 InstanceOpLowering::matchAndRewrite(InstanceOp msftInst, OpAdaptor adaptor,
58  ConversionPatternRewriter &rewriter) const {
59  Operation *referencedModule = msftInst.getReferencedModule();
60  if (!referencedModule)
61  return rewriter.notifyMatchFailure(msftInst,
62  "Could not find referenced module");
63  if (!hw::isAnyModule(referencedModule))
64  return rewriter.notifyMatchFailure(
65  msftInst, "Referenced module was not an HW module");
66  auto hwInst = rewriter.create<hw::InstanceOp>(
67  msftInst.getLoc(), referencedModule, msftInst.instanceNameAttr(),
68  SmallVector<Value>(adaptor.getOperands().begin(),
69  adaptor.getOperands().end()),
70  msftInst.parameters().getValueOr(ArrayAttr()), msftInst.sym_nameAttr());
71  rewriter.replaceOp(msftInst, hwInst.getResults());
72  return success();
73 }

93 ModuleOpLowering::matchAndRewrite(MSFTModuleOp mod, OpAdaptor adaptor,
94  ConversionPatternRewriter &rewriter) const {
95  if (mod.body().empty()) {
96  std::string comment;
97  llvm::raw_string_ostream(comment)
98  << "// Module not generated: \"" << mod.getName() << "\" params "
99  << mod.parameters();
100  // TODO: replace this with proper comment op when it's created.
101  rewriter.replaceOpWithNewOp<sv::VerbatimOp>(mod, comment);
102  return success();
103  }
104 
105  auto hwmod = rewriter.replaceOpWithNewOp<hw::HWModuleOp>(
106  mod, mod.getNameAttr(), mod.getPorts());
107  rewriter.eraseBlock(hwmod.getBodyBlock());

109  hwmod.getBody().end());
110 
111  auto opOutputFile = mod.fileName();
112  if (opOutputFile) {
113  auto outputFileAttr = hw::OutputFileAttr::getFromFilename(
114  rewriter.getContext(), *opOutputFile, false, true);
115  hwmod->setAttr("output_file", outputFileAttr);
116  } else if (!outputFile.empty()) {
117  auto outputFileAttr = hw::OutputFileAttr::getFromFilename(
118  rewriter.getContext(), outputFile, false, true);
119  hwmod->setAttr("output_file", outputFileAttr);
120  }
121 
122  return success();

214 void LowerToHWPass::runOnOperation() {
215  auto top = getOperation();
216  auto *ctxt = &getContext();
217 
218  // Traverse MSFT location attributes and export the required Tcl into
219  // templated `sv::VerbatimOp`s with symbolic references to the instance paths.
220  for (auto moduleName : tops) {
221  auto hwmod = top.lookupSymbol<msft::MSFTModuleOp>(moduleName);
222  if (!hwmod)
223  continue;
224  if (failed(exportQuartusTcl(hwmod, tclFile)))
225  return signalPassFailure();
226  }
227 
228  // The `hw::InstanceOp` (which `msft::InstanceOp` lowers to) convenience
229  // builder gets its argNames and resultNames from the `hw::HWModuleOp`. So we
230  // have to lower `msft::MSFTModuleOp` before we lower `msft::InstanceOp`.
231 
232  // Convert everything except instance ops first.
233 
234  ConversionTarget target(*ctxt);
235  target.addIllegalOp<MSFTModuleOp, MSFTModuleExternOp, OutputOp>();
236  target.addLegalDialect<hw::HWDialect>();
237  target.addLegalDialect<sv::SVDialect>();
238  target.addDynamicallyLegalOp<hw::GlobalRefOp>([](hw::GlobalRefOp op) {
239  for (auto attr : op->getAttrs())
240  if (isa<MSFTDialect>(attr.getValue().getDialect()))
241  return false;
242  return true;
243  });
244 
245  RewritePatternSet patterns(ctxt);
246  patterns.insert<ModuleOpLowering>(ctxt, verilogFile);
247  patterns.insert<ModuleExternOpLowering>(ctxt, verilogFile);
248  patterns.insert<OutputOpLowering>(ctxt);
249  patterns.insert<GlobalRefOpLowering>(ctxt);
250 
251  if (failed(applyPartialConversion(top, target, std::move(patterns))))
252  signalPassFailure();
253 
254  // Then, convert the InstanceOps
255  target.addIllegalOp<msft::InstanceOp>();
256  RewritePatternSet instancePatterns(ctxt);
257  instancePatterns.insert<InstanceOpLowering>(ctxt);
258  if (failed(applyPartialConversion(top, target, std::move(instancePatterns))))
259  signalPassFailure();
260 
261  // Finally, legalize the rest of the MSFT dialect.
262  target.addIllegalDialect<MSFTDialect>();
263  RewritePatternSet finalPatterns(ctxt);
264  finalPatterns.insert<PhysicalRegionOpLowering>(ctxt);
265  if (failed(applyPartialConversion(top, target, std::move(finalPatterns))))
266  signalPassFailure();
267 }

311 /// Update all the instantiations of 'mod' to match the port list. For any
312 /// output ports which survived, automatically map the result according to
313 /// `newToOldResultMap`. Calls 'getOperandsFunc' with the new instance op, the
314 /// old instance op, and expects the operand vector to return filled.
315 /// `getOperandsFunc` can (and often does) modify other operations. The update
316 /// call deletes the original instance op, so all references are invalidated
317 /// after this call.
318 void PassCommon::updateInstances(
319  MSFTModuleOp mod, ArrayRef<unsigned> newToOldResultMap,
320  llvm::function_ref<void(InstanceOp, InstanceOp, SmallVectorImpl<Value> &)>
321  getOperandsFunc) {
322  SmallVector<InstanceOp, 1> newInstances;
323  for (InstanceOp inst : moduleInstantiations[mod]) {
324  assert(inst->getParentOp());
325  OpBuilder b(inst);
326  auto newInst =
327  b.create<InstanceOp>(inst.getLoc(), mod.getType().getResults(),
328  inst.getOperands(), inst->getAttrs());
329  for (size_t portNum = 0, e = newToOldResultMap.size(); portNum < e;
330  ++portNum) {
331  assert(portNum < newInst.getNumResults());
332  inst.getResult(newToOldResultMap[portNum])
333  .replaceAllUsesWith(newInst.getResult(portNum));
334  }
335 
336  SmallVector<Value> newOperands;
337  getOperandsFunc(newInst, inst, newOperands);
338  newInst->setOperands(newOperands);
339 
340  newInstances.push_back(newInst);
341  inst->dropAllUses();
342  inst->erase();
343  }
344  moduleInstantiations[mod].swap(newInstances);
345 }

347 // Run a post-order DFS.
348 void PassCommon::getAndSortModulesVisitor(MSFTModuleOp mod,
349  SmallVectorImpl<MSFTModuleOp> &mods,
350  DenseSet<MSFTModuleOp> &modsSeen) {
351  if (modsSeen.contains(mod))
352  return;
353  modsSeen.insert(mod);
354 
355  mod.walk([&](InstanceOp inst) {
356  Operation *modOp = topLevelSyms.getDefinition(inst.moduleNameAttr());
357  auto mod = dyn_cast_or_null<MSFTModuleOp>(modOp);
358  if (!mod)
359  return;
360  moduleInstantiations[mod].push_back(inst);
361  getAndSortModulesVisitor(mod, mods, modsSeen);
362  });
363 
364  mods.push_back(mod);
365 }

367 void PassCommon::getAndSortModules(ModuleOp topMod,
368  SmallVectorImpl<MSFTModuleOp> &mods) {
369  // Add here _before_ we go deeper to prevent infinite recursion.
370  DenseSet<MSFTModuleOp> modsSeen;
371  mods.clear();
372  moduleInstantiations.clear();
373  topMod.walk(
374  [&](MSFTModuleOp mod) { getAndSortModulesVisitor(mod, mods, modsSeen); });
375 }

404 void PartitionPass::runOnOperation() {
405  ModuleOp outerMod = getOperation();
406  populateSymbolCache(outerMod);
407 
408  // Get a properly sorted list, then partition the mods in order.
409  SmallVector<MSFTModuleOp, 64> sortedMods;
410  getAndSortModules(outerMod, sortedMods);
411  for (auto mod : sortedMods)
412  partition(mod);

415 void PartitionPass::markWireOps(Operation *taggedOp) {
416  auto partRef =
417  taggedOp->getAttrOfType<SymbolRefAttr>("targetDesignPartition");
418  if (!partRef)
419  return;
420  SmallVector<Operation *, 8> opQueue;
421  opQueue.push_back(taggedOp);
422 
423  while (!opQueue.empty()) {
424  Operation *op = opQueue.back();
425  opQueue.pop_back();
426 
427  for (auto *user : op->getUsers()) {
428  if (!isWireManipulationOp(user) || user->hasAttr("targetDesignPartition"))
429  continue;
430  user->setAttr("targetDesignPartition", partRef);
431  opQueue.push_back(user);
432  }
433 
434  for (auto operValue : op->getOperands()) {
435  Operation *defOp = operValue.getDefiningOp();
436  if (!defOp || !isWireManipulationOp(defOp) ||
437  defOp->hasAttr("targetDesignPartition"))
438  continue;
439  defOp->setAttr("targetDesignPartition", partRef);
440  opQueue.push_back(defOp);
441  }
442  }
443 }

445 void PartitionPass::partition(MSFTModuleOp mod) {
446  DenseMap<StringAttr, SmallVector<Operation *, 1>> localPartMembers;
447  SmallVector<Operation *, 64> nonLocalTaggedOps;
448 
449  mod.walk(&markWireOps);
450 
451  mod.walk([&](Operation *op) {
452  auto partRef = op->getAttrOfType<SymbolRefAttr>("targetDesignPartition");
453  if (!partRef)
454  return;
455 
456  if (partRef.getRootReference() == SymbolTable::getSymbolName(mod))
457  localPartMembers[partRef.getLeafReference()].push_back(op);
458  else
459  nonLocalTaggedOps.push_back(op);
460  });
461 
462  bubbleUp(mod, nonLocalTaggedOps);
463 
464  for (auto part :
465  llvm::make_early_inc_range(mod.getOps<DesignPartitionOp>())) {
466  auto usersIter = localPartMembers.find(part.sym_nameAttr());
467  if (usersIter != localPartMembers.end())
468  this->partition(part, usersIter->second);
469  part.erase();
470  }
471 }

504 /// Heuristics to get the output name.
505 static StringRef getResultName(OpResult res, const hw::SymbolCache &syms,
506  std::string &buff) {
507  Operation *op = res.getDefiningOp();
508  if (auto inst = dyn_cast<InstanceOp>(op)) {
509  Operation *modOp = syms.getDefinition(inst.moduleNameAttr());
510  assert(modOp && "Invalid IR");
511  assert(isAnyModule(modOp) && "Instance must point to a module");
512  hw::ModulePortInfo ports = getModulePortInfo(modOp);
513  return ports.outputs[res.getResultNumber()].name;
514  }
515  if (auto asmInterface = dyn_cast<mlir::OpAsmOpInterface>(op)) {
516  StringRef retName;
517  asmInterface.getAsmResultNames([&](Value v, StringRef name) {
518  if (v == res && !name.empty())
519  retName = StringAttr::get(op->getContext(), name).getValue();
520  });
521  if (!retName.empty())
522  return retName;
523  }
524  if (auto constOp = dyn_cast<hw::ConstantOp>(op)) {
525  buff.clear();
526  llvm::raw_string_ostream(buff) << "c" << constOp.value();
527  return buff;
528  }
529 
530  if (res.getDefiningOp()->getNumResults() == 1)
531  return {};
532 
533  // Fallback. Not ideal.
534  buff.clear();
535  llvm::raw_string_ostream(buff) << "out" << res.getResultNumber();
536  return buff;
537 }

560 void PartitionPass::bubbleUp(MSFTModuleOp mod, ArrayRef<Operation *> ops) {
561  // This particular implementation is _very_ sensitive to iteration order. It
562  // assumes that the order in which the ops, operands, and results are the same
563  // _every_ time it runs through them. Doing this saves on bookkeeping.
564  auto *ctxt = mod.getContext();
565  FunctionType origType = mod.getType();
566  std::string nameBuffer;
567 
568  //*************
569  // Figure out all the new ports 'mod' is going to need. The outputs need to
570  // know where they're being driven from, which'll be the outputs of 'ops'.
571  SmallVector<std::pair<StringAttr, Type>, 64> newInputs;
572  SmallVector<std::pair<StringAttr, Value>, 64> newOutputs;
573  SmallVector<Type, 64> newResTypes;
574 
575  for (Operation *op : ops) {
576  StringRef opName = ::getOpName(op);
577  for (OpResult res : op->getOpResults()) {
578  StringRef name = getResultName(res, topLevelSyms, nameBuffer);
579  newInputs.push_back(std::make_pair(
580  StringAttr::get(ctxt, opName + (name.empty() ? "" : "." + name)),
581  res.getType()));
582  }
583  for (OpOperand &oper : op->getOpOperands()) {
584  StringRef name = getOperandName(oper, topLevelSyms, nameBuffer);
585  newOutputs.push_back(std::make_pair(
586  StringAttr::get(ctxt, opName + (name.empty() ? "" : "." + name)),
587  oper.get()));
588  newResTypes.push_back(oper.get().getType());
589  }
590  }
591 
592  //*************
593  // Add them to 'mod' and replace all of the old 'ops' results with the new
594  // ports.
595  SmallVector<BlockArgument> newBlockArgs = mod.addPorts(newInputs, newOutputs);
596  size_t blockArgNum = 0;
597  for (Operation *op : ops)
598  for (OpResult res : op->getResults())
599  res.replaceAllUsesWith(newBlockArgs[blockArgNum++]);
600 
601  //*************
602  // For all of the instantiation sites (for 'mod'):
603  // - Create a new instance with the correct result types.
604  // - Clone in 'ops'.
605  // - Construct the new instance operands from the old ones + the cloned
606  // ops' results.
607  SmallVector<unsigned> resValues;
608  for (size_t i = 0, e = origType.getNumInputs(); i < e; ++i)
609  resValues.push_back(i);
610  auto cloneOpsGetOperands = [&](InstanceOp newInst, InstanceOp oldInst,
611  SmallVectorImpl<Value> &newOperands) {
612  OpBuilder b(newInst);
613 
614  size_t resultNum = origType.getNumResults();
615  auto oldOperands = oldInst->getOperands();
616  newOperands.append(oldOperands.begin(), oldOperands.end());
617  for (Operation *op : ops) {
618  BlockAndValueMapping map;
619  for (Value oper : op->getOperands())
620  map.map(oper, newInst->getResult(resultNum++));
621  Operation *newOp = b.insert(op->clone(map));
622  for (Value res : newOp->getResults())
623  newOperands.push_back(res);
624  setEntityName(newOp, oldInst.getName() + "." + ::getOpName(op));
625  }
626  };
627  updateInstances(mod, resValues, cloneOpsGetOperands);
628 
629  //*************
630  // Done.
631  for (Operation *op : ops)
632  op->erase();

635 void PartitionPass::partition(DesignPartitionOp partOp,
636  ArrayRef<Operation *> toMove) {
637 
638  auto *ctxt = partOp.getContext();
639  auto loc = partOp.getLoc();
640  std::string nameBuffer;
641 
642  //*************
643  // Determine the partition module's interface. Keep some bookkeeping around.
644  SmallVector<hw::PortInfo> inputPorts;
645  SmallVector<hw::PortInfo> outputPorts;
646  SmallVector<Value, 64> partInstInputs;
647  SmallVector<Value, 64> partInstOutputs;
648 
649  // Handle module-like operations. Not strictly necessary, but we can base the
650  // new portnames on the portnames of the instance being moved and the instance
651  // name.
652  auto addModuleLike = [&](Operation *inst, Operation *modOp) {
653  hw::ModulePortInfo modPorts = getModulePortInfo(modOp);
654  StringRef name = ::getOpName(inst);
655 
656  for (auto port : modPorts.inputs) {
657  partInstInputs.push_back(inst->getOperand(port.argNum));
658  inputPorts.push_back(hw::PortInfo{
659  /*name*/ StringAttr::get(ctxt, name + "." + port.name.getValue()),
660  /*direction*/ hw::PortDirection::INPUT,
661  /*type*/ port.type,
662  /*argNum*/ inputPorts.size()});
663  }
664 
665  for (auto port : modPorts.outputs) {
666  partInstOutputs.push_back(inst->getResult(port.argNum));
667  outputPorts.push_back(hw::PortInfo{
668  /*name*/ StringAttr::get(ctxt, name + "." + port.name.getValue()),
669  /*direction*/ hw::PortDirection::OUTPUT,
670  /*type*/ port.type,
671  /*argNum*/ outputPorts.size()});
672  }
673  };
674  // Handle all other operators.
675  auto addOther = [&](Operation *op) {
676  StringRef name = ::getOpName(op);
677 
678  for (OpOperand &oper : op->getOpOperands()) {
679  inputPorts.push_back(hw::PortInfo{
680  /*name*/ StringAttr::get(
681  ctxt,
682  name + "." + getOperandName(oper, topLevelSyms, nameBuffer)),
683  /*direction*/ hw::PortDirection::INPUT,
684  /*type*/ oper.get().getType(),
685  /*argNum*/ inputPorts.size()});
686  partInstInputs.push_back(oper.get());
687  }
688 
689  for (OpResult res : op->getOpResults()) {
690  StringRef resName = getResultName(res, topLevelSyms, nameBuffer);
691  outputPorts.push_back(
692  hw::PortInfo{/*name*/ StringAttr::get(
693  ctxt, name + (resName.empty() ? "" : "." + resName)),
694  /*direction*/ hw::PortDirection::OUTPUT,
695  /*type*/ res.getType(),
696  /*argNum*/ outputPorts.size()});
697  partInstOutputs.push_back(res);
698  }
699  };
700 
701  // Aggregate the args/results into partition module ports.
702  for (Operation *op : toMove) {
703  if (auto inst = dyn_cast<InstanceOp>(op)) {
704  Operation *modOp = topLevelSyms.getDefinition(inst.moduleNameAttr());
705  assert(modOp && "Module instantiated should exist. Verifier should have "
706  "caught this.");
707 
708  if (isAnyModule(modOp))
709  addModuleLike(inst, modOp);
710  } else {
711  addOther(op);
712  }
713  }
714 
715  //*************
716  // Construct the partition module and replace the design partition op.
717 
718  // Build the module.
719  hw::ModulePortInfo modPortInfo(inputPorts, outputPorts);
720  auto partMod =
721  OpBuilder::atBlockEnd(getOperation().getBody())
722  .create<MSFTModuleOp>(loc, partOp.verilogNameAttr(), modPortInfo,
723  ArrayRef<NamedAttribute>{});
724  Block *partBlock = partMod.getBodyBlock();
725  partBlock->clear();
726  auto partBuilder = OpBuilder::atBlockEnd(partBlock);
727 
728  // Replace partOp with an instantion of the partition.
729  SmallVector<Type> instRetTypes(
730  llvm::map_range(partInstOutputs, [](Value v) { return v.getType(); }));
731  auto partInst = OpBuilder(partOp).create<InstanceOp>(
732  loc, instRetTypes, partOp.getNameAttr(),
733  SymbolTable::getSymbolName(partMod), partInstInputs, ArrayAttr(),
734  SymbolRefAttr());
735 
736  //*************
737  // Move the operations!
738 
739  // Map the original operation's inputs to block arguments.
740  mlir::BlockAndValueMapping mapping;
741  assert(partInstInputs.size() == partBlock->getNumArguments());
742  for (size_t argNum = 0, e = partInstInputs.size(); argNum < e; ++argNum) {
743  mapping.map(partInstInputs[argNum], partBlock->getArgument(argNum));
744  }
745 
746  // Since the same value can map to multiple outputs, compute the 1-N mapping
747  // here.
748  DenseMap<Value, SmallVector<int, 1>> resultOutputConnections;
749  for (size_t outNum = 0, e = partInstOutputs.size(); outNum < e; ++outNum)
750  resultOutputConnections[partInstOutputs[outNum]].push_back(outNum);
751 
752  // Hold the block outputs.
753  SmallVector<Value, 64> outputs(partInstOutputs.size(), Value());
754 
755  // Clone the ops into the partition block. Map the results into the module
756  // outputs.
757  for (Operation *op : toMove) {
758  Operation *newOp = partBuilder.insert(op->clone(mapping));
759  newOp->removeAttr("targetDesignPartition");
760  for (size_t resNum = 0, e = op->getNumResults(); resNum < e; ++resNum)
761  for (int outputNum : resultOutputConnections[op->getResult(resNum)])
762  outputs[outputNum] = newOp->getResult(resNum);
763  }
764  partBuilder.create<OutputOp>(loc, outputs);
765 
766  // Replace original ops' outputs with partition outputs.
767  assert(partInstOutputs.size() == partInst.getNumResults());
768  for (size_t resNum = 0, e = partInstOutputs.size(); resNum < e; ++resNum)
769  partInstOutputs[resNum].replaceAllUsesWith(partInst.getResult(resNum));
770 
771  for (Operation *op : toMove)
772  op->erase();
773 }

806 /// Push up any wires which are simply passed-through.
807 void WireCleanupPass::bubbleWiresUp(MSFTModuleOp mod) {
808  Block *body = mod.getBodyBlock();
809  Operation *terminator = body->getTerminator();
810  hw::ModulePortInfo ports = mod.getPorts();
811 
812  // Find all "passthough" internal wires, filling 'inputPortsToRemove' as a
813  // side-effect.
814  DenseMap<Value, hw::PortInfo> passThroughs;
815  llvm::BitVector inputPortsToRemove(ports.inputs.size());
816  for (hw::PortInfo inputPort : ports.inputs) {
817  BlockArgument portArg = body->getArgument(inputPort.argNum);
818  bool removePort = true;
819  for (OpOperand user : portArg.getUsers()) {
820  if (user.getOwner() == terminator)
821  passThroughs[portArg] = inputPort;
822  else
823  removePort = false;
824  }
825  if (removePort)
826  inputPortsToRemove.set(inputPort.argNum);
827  }
828 
829  // Find all output ports which we can remove. Fill in 'outputToInputIdx' to
830  // help rewire instantiations later on.
831  DenseMap<unsigned, unsigned> outputToInputIdx;
832  llvm::BitVector outputPortsToRemove(ports.outputs.size());
833  for (hw::PortInfo outputPort : ports.outputs) {
834  assert(outputPort.argNum < terminator->getNumOperands() && "Invalid IR");
835  Value outputValue = terminator->getOperand(outputPort.argNum);
836  auto inputNumF = passThroughs.find(outputValue);
837  if (inputNumF == passThroughs.end())
838  continue;
839  hw::PortInfo inputPort = inputNumF->second;
840  outputToInputIdx[outputPort.argNum] = inputPort.argNum;
841  outputPortsToRemove.set(outputPort.argNum);
842  }
843 
844  // Use MSFTModuleOp's `removePorts` method to remove the ports. It returns a
845  // mapping of the new output port to old output port indices to assist in
846  // updating the instantiations later on.
847  auto newToOldResult =
848  mod.removePorts(inputPortsToRemove, outputPortsToRemove);
849 
850  // Update the instantiations.
851  auto setPassthroughsGetOperands = [&](InstanceOp newInst, InstanceOp oldInst,
852  SmallVectorImpl<Value> &newOperands) {
853  // Re-map the passthrough values around the instance.
854  for (auto idxPair : outputToInputIdx) {
855  size_t outputPortNum = idxPair.first;
856  assert(outputPortNum <= oldInst.getNumResults());
857  size_t inputPortNum = idxPair.second;
858  assert(inputPortNum <= oldInst.getNumOperands());
859  oldInst.getResult(outputPortNum)
860  .replaceAllUsesWith(oldInst.getOperand(inputPortNum));
861  }
862  // Use a sort-merge-join approach to figure out the operand mapping on the
863  // fly.
864  for (size_t operNum = 0, e = oldInst.getNumOperands(); operNum < e;
865  ++operNum)
866  if (!inputPortsToRemove.test(operNum))
867  newOperands.push_back(oldInst.getOperand(operNum));
868  };
869  updateInstances(mod, newToOldResult, setPassthroughsGetOperands);
870 }

872 /// Sink all the instance connections which are loops.
873 void WireCleanupPass::sinkWiresDown(MSFTModuleOp mod) {
874  auto instantiations = moduleInstantiations[mod];
875  // TODO: remove this limitation. This would involve looking at the common
876  // loopbacks for all the instances.
877  if (instantiations.size() != 1)
878  return;
879  InstanceOp inst = instantiations[0];
880 
881  // Find all the "loopback" connections in the instantiation. Populate
882  // 'inputToOutputLoopback' with a mapping of input port to output port which
883  // drives it. Populate 'resultsToErase' with output ports which only drive
884  // input ports.
885  DenseMap<unsigned, unsigned> inputToOutputLoopback;
886  llvm::BitVector resultsToErase(inst.getNumResults());
887  for (unsigned resNum = 0, e = inst.getNumResults(); resNum < e; ++resNum) {
888  bool allLoops = true;
889  for (auto &use : inst.getResult(resNum).getUses()) {
890  if (use.getOwner() != inst.getOperation())
891  allLoops = false;
892  else
893  inputToOutputLoopback[use.getOperandNumber()] = resNum;
894  }
895  if (allLoops)
896  resultsToErase.set(resNum);
897  }
898 
899  // Add internal connections to replace the instantiation's loop back
900  // connections.
901  Block *body = mod.getBodyBlock();
902  Operation *terminator = body->getTerminator();
903  llvm::BitVector argsToErase(body->getNumArguments());
904  for (auto resOper : inputToOutputLoopback) {
905  body->getArgument(resOper.first)
906  .replaceAllUsesWith(terminator->getOperand(resOper.second));
907  argsToErase.set(resOper.first);
908  }
909 
910  // Remove the ports.
911  SmallVector<unsigned> newToOldResultMap =
912  mod.removePorts(argsToErase, resultsToErase);
913  // and update the instantiations.
914  auto getOperands = [&](InstanceOp newInst, InstanceOp oldInst,
915  SmallVectorImpl<Value> &newOperands) {
916  // Use sort-merge-join to compute the new operands;
917  for (unsigned argNum = 0, e = oldInst.getNumOperands(); argNum < e;
918  ++argNum)
919  if (!argsToErase.test(argNum))
920  newOperands.push_back(oldInst.getOperand(argNum));
921  };
922  updateInstances(mod, newToOldResultMap, getOperands);
923 }
